Code Examples

Copy-paste into your agent

JavaScript / Node.js

const API_BASE = 'https://your-middleware.com';

// 1. Register your agent
async function registerAgent(name, walletAddress) {
  const res = await fetch(`${API_BASE}/v1/agents/register`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ 
      name, 
      wallet_address: walletAddress 
    })
  });
  const { agent } = await res.json();
  return agent.uid; // e.g., "KA-abc123xyz"
}

// 2. Create payment link
async function createPaymentLink(agentUid, amount, description) {
  const res = await fetch(`${API_BASE}/v1/payments/create-link`, {
    method: 'POST',
    headers: { 
      'Content-Type': 'application/json',
      'X-Agent-UID': agentUid
    },
    body: JSON.stringify({ 
      amount, 
      description 
    })
  });
  const { payment } = await res.json();
  return payment.link; // Send this to customer
}

Python

import requests

API_BASE = 'https://your-middleware.com'

# 1. Register your agent
def register_agent(name: str, wallet_address: str) -> str:
    res = requests.post(f'{API_BASE}/v1/agents/register', json={
        'name': name,
        'wallet_address': wallet_address
    })
    agent = res.json()['agent']
    return agent['uid']  # e.g., "KA-abc123xyz"

# 2. Create payment link
def create_payment_link(agent_uid: str, amount: float, description: str) -> str:
    res = requests.post(
        f'{API_BASE}/v1/payments/create-link',
        headers={'X-Agent-UID': agent_uid},
        json={'amount': amount, 'description': description}
    )
    return res.json()['payment']['link']

# 3. Webhook handler (Flask example)

    return jsonify({'ok': True})
